118 void GetMainInstAndWin (HINSTANCE * hMainProgInst, HWND * hMainProgWindow)
119 
120 {
121  #if defined EXTERN_WINDOW
122  *hMainProgWindow= EXTERN_WINDOW;
123  #else
124  *hMainProgWindow= NULL; // wird bei Bedarf spaeter erzeugt
125  #endif
126 
127  #if defined EXTERN_INSTANCE
128  *hMainProgInst= EXTERN_INSTANCE;
129  #else
130  *hMainProgInst= NULL;
131  #endif
132 
133  if (*hMainProgInst == NULL) {
134  #if defined EXTERN_WINDOW
135  if (EXTERN_WINDOW != NULL ) { // Hauptprogramm besitzt (bekanntes) Fenster
136  #if defined __WATCOMC__ // Watcom Default Window System 16/32 bit
137  #if (!defined(__WIN32__) && !defined(_WIN32))
138  *hMainProgInst= (HINSTANCE)GetWindowWord(EXTERN_WINDOW, GWW_HINSTANCE);
139  #else // Watcom ohne 64bit Windows
140  *hMainProgInst= (HINSTANCE)GetWindowLong(EXTERN_WINDOW, GWL_HINSTANCE);
141  #endif
142  #else // alle anderen Compiler ohne 16bit Windows
143  #if (!defined(_WIN64)) // 32 bit
144  *hMainProgInst= (HINSTANCE)GetWindowLong(EXTERN_WINDOW, GWL_HINSTANCE);
145  #else // 64 bit
146  *hMainProgInst= (HINSTANCE)GetWindowLongPtr(EXTERN_WINDOW, GWLP_HINSTANCE);
147  #endif
148  #endif
149  } else { // kein offenes Fenster, z.B. Watcom-Consolenanwendung
150  *hMainProgInst= GetModuleHandle (NULL);
151  }
152  #else // kein Fenster ermittelbar
153  *hMainProgInst= GetModuleHandle (NULL);
154  #endif
155  }
156 }
