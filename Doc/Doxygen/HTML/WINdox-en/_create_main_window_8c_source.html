87  wincl.hInstance = *hMainProgInst;
88  wincl.lpszClassName = szClassName;
89  wincl.lpfnWndProc = DefWindowProc; /* keine eigene Windowsroutine */

97  wincl.hbrBackground = (HBRUSH) COLOR_BACKGROUND;
98 
99  /* Register the window class. Fail: most probable UNICODE on win98 */
100  if (!RegisterClass (&wincl)) {
101  #if defined(__WIN32__) || defined(_WIN32)
102  ErrorCode= GetLastError(); // win32-Funktion
103 // if (ErrorCode == ERROR_CLASS_ALREADY_EXISTS) {

106  FormatMessage(
107  FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
108  NULL,

111  (LPTSTR) &lpMsgBuf,

116  LocalFree( lpMsgBuf ); // Free the buffer

122  return;
123  }
124 
125  /* The class is registered, let's create the program */
126  *hMainProgWindow = CreateWindow (
127  szClassName, // Classname
128  szWinName, // Title Text

130  CW_USEDEFAULT, // Windows decides the position

139  ShowWindow (*hMainProgWindow, SW_SHOW);
