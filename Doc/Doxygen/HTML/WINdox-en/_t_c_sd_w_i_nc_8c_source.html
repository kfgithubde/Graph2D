708 bool ClipLineStart (FTNINT ix1, FTNINT iy1, FTNINT ix2, FTNINT iy2,
709  FTNINT *isx, FTNINT *isy)
710 /* ClipLineStart=true: isx,isy Startpunkt; =false: Linie nicht zeichnen */
711 {
712  if (ClippingNotActive) {
713  *isx= ix1; *isy= iy1;
714  return true;
715  }
716 
717  if (ix1 < TKTRNX.kminsx) { /* Start links vom Fenster */
718  if (ix2 < TKTRNX.kminsx) return false;
719  *isy= iy1+((TKTRNX.kminsx-ix1) * (iy2-iy1)) / (ix2-ix1);
720  if ((TKTRNX.kminsy <= *isy) && (TKTRNX.kmaxsy >= *isy)) {
721  *isx= TKTRNX.kminsx;
722  return true;
723  }
724  if (iy1 == iy2) return false;
725  if (((ix2-ix1)*(iy2-iy1)) >= 0) { /* Steigung positiv */
726  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
727  *isy= TKTRNX.kminsy;
728  } else {
729  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
730  *isy= TKTRNX.kmaxsy;
731  }
732  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
733  return true;
734 
735  } else if (ix1 > TKTRNX.kmaxsx) { /* Start rechts vom Fenster */
736  if (ix2 > TKTRNX.kmaxsx) return false;
737  *isy= iy1+((TKTRNX.kmaxsx-ix1) * (iy2-iy1)) / (ix2-ix1);
738  if ((TKTRNX.kminsy <= *isy) && (TKTRNX.kmaxsy >= *isy)) {
739  *isx= TKTRNX.kmaxsx;
740  return true;
741  }
742  if (iy1 == iy2) return false;
743  if (((ix2-ix1)*(iy2-iy1)) >= 0) { /* Steigung positiv */
744  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
745  *isy= TKTRNX.kmaxsy;
746  } else {
747  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
748  *isy= TKTRNX.kminsy;
749  }
750  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
751  return true;
752 
753  } else if (iy1 < TKTRNX.kminsy) { /* Start unter dem Fenster */
754  if (iy2 < TKTRNX.kminsy) return false;
755  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
756  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
757  *isy= TKTRNX.kminsy;
758  return true;
759 
760  } else if (iy1 > TKTRNX.kmaxsy) { /* Start ueber dem Fenster */
761  if (iy2 > TKTRNX.kmaxsy) return false;
762  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
763  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
764  *isy= TKTRNX.kmaxsy;
765  return true;
766 
767  }
768  *isx= ix1; /* Startpunkt liegt im Fenster */
769  *isy= iy1;
770  return true;
