299 bool ClipLineStart (FTNINT ix1, FTNINT iy1, FTNINT ix2, FTNINT iy2,
300  FTNINT *isx, FTNINT *isy)
301 /* ClipLineStart=true: isx,isy Startpunkt; =false: Linie nicht zeichnen */
302 {
303  if (ClippingNotActive) {
304  *isx= ix1; *isy= iy1;
305  return true;
306  }
307 
308  if (ix1 < TKTRNX.kminsx) { /* Start links vom Fenster */
309  if (ix2 < TKTRNX.kminsx) return false;
310  *isy= iy1+((TKTRNX.kminsx-ix1) * (iy2-iy1)) / (ix2-ix1);
311  if ((TKTRNX.kminsy <= *isy) && (TKTRNX.kmaxsy >= *isy)) {
312  *isx= TKTRNX.kminsx;
313  return true;
314  }
315  if (iy1 == iy2) return false;
316  if (((ix2-ix1)*(iy2-iy1)) >= 0) { /* Steigung positiv */
317  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
318  *isy= TKTRNX.kminsy;
319  } else {
320  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
321  *isy= TKTRNX.kmaxsy;
322  }
323  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
324  return true;
325 
326  } else if (ix1 > TKTRNX.kmaxsx) { /* Start rechts vom Fenster */
327  if (ix2 > TKTRNX.kmaxsx) return false;
328  *isy= iy1+((TKTRNX.kmaxsx-ix1) * (iy2-iy1)) / (ix2-ix1);
329  if ((TKTRNX.kminsy <= *isy) && (TKTRNX.kmaxsy >= *isy)) {
330  *isx= TKTRNX.kmaxsx;
331  return true;
332  }
333  if (iy1 == iy2) return false;
334  if (((ix2-ix1)*(iy2-iy1)) >= 0) { /* Steigung positiv */
335  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
336  *isy= TKTRNX.kmaxsy;
337  } else {
338  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
339  *isy= TKTRNX.kminsy;
340  }
341  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
342  return true;
343 
344  } else if (iy1 < TKTRNX.kminsy) { /* Start unter dem Fenster */
345  if (iy2 < TKTRNX.kminsy) return false;
346  *isx= ix1+ ((TKTRNX.kminsy-iy1)*(ix2-ix1))/(iy2-iy1);
347  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
348  *isy= TKTRNX.kminsy;
349  return true;
350 
351  } else if (iy1 > TKTRNX.kmaxsy) { /* Start ueber dem Fenster */
352  if (iy2 > TKTRNX.kmaxsy) return false;
353  *isx= ix1+ ((TKTRNX.kmaxsy-iy1)*(ix2-ix1))/(iy2-iy1);
354  if ((*isx > TKTRNX.kmaxsx) || (*isx < TKTRNX.kminsx)) return false;
355  *isy= TKTRNX.kmaxsy;
356  return true;
357 
358  }
359  *isx= ix1; /* Startpunkt liegt im Fenster */
360  *isy= iy1;
361  return true;

366 void DrawHiResDashLine (FTNINT ix,FTNINT iy, FTNINT ix2,FTNINT iy2,FTNINT *iMask)
367 {
368 FTNINT ixx,iyy, ixx2,iyy2;
369 float xx,yy, dx,dy, dLin,dBlank;
370 
371  if (*iMask <= 0) {
372  dLin= 10., dBlank=0.; // solid
373  } else if (*iMask == 1) {
374  dLin= 1.; dBlank=1.; // dotted
375  } else if (*iMask == 2) {
376  dLin= 3.; dBlank=1.; // substitute dashed-dotted
377  } else if (*iMask == 3) {
378  dLin= 3.; dBlank=3.; // dashed
379  } else {
380  dLin= 3., dBlank=3.; // unrecognized -> dashed
381  }
382 
383  if (abs(ix2-ix) >= abs(iy2-iy)) {
384  dx= ix2 >= ix ? 3. : -3.;
385  dy= ((float)(iy2-iy))/((float)(ix2-ix))*dx;
386 
387  xx= (float)ix; yy= (float)iy;
388  while (dx != 0.) {
389  ixx= (FTNINT) xx; iyy= (FTNINT) yy;
390  ixx2=(FTNINT) (xx+dLin*dx); iyy2=(FTNINT) (yy+dLin*dy);
391  xx+= (dLin+dBlank)*dx; yy+= (dLin+dBlank)*dy;
392  if ( ((dx>=0.) && ((FTNINT)xx>=ix2) )
393  || ((dx<=0.) && ((FTNINT)xx<=ix2) ) ) {
394  ixx2= ix2; iyy2= iy2;
395  dx= 0.;
396  }
397  SDL_RenderDrawLine(TCSrenderer, HiResX(ixx),HiResY(TEK_YMAX-iyy),
398  HiResX(ixx2),HiResY(TEK_YMAX-iyy2));
399  }
400 
401  } else {
402  dy= iy2 >= iy ? 3. : -3.;
403  dx= ((float)(ix2-ix))/((float)(iy2-iy))*dy;
404 
405  xx= (float)ix; yy= (float)iy;
406  while (dy != 0.) {
407  ixx= (FTNINT) xx; iyy= (FTNINT) yy;
408  ixx2=(FTNINT) (xx+dLin*dx); iyy2=(FTNINT) (yy+dLin*dy);
409  xx+= (dLin+dBlank)*dx; yy+= (dLin+dBlank)*dy;
410  if ( ((dy>=0.) && ((FTNINT)yy>=iy2) )
411  || ((dy<=0.) && ((FTNINT)yy<=iy2) ) ) {
412  ixx2= ix2; iyy2= iy2;
413  dy= 0.;
414  }
415  SDL_RenderDrawLine(TCSrenderer, HiResX(ixx), HiResY(TEK_YMAX-iyy),
